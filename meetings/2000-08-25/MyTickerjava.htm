// MyTicker Applet



import java.awt.*;

import java.applet.Applet;



public class MyTicker extends Applet {

   String s;

   

   public void init() {

      s = "Hello UMSL";

   }

   

   public void paint(Graphics g) {

      try {

         while (true) {

            for (int x = 300; (x+s.length())>0;x--) {

               g.drawString(s,x,20);

               Thread.sleep(150);

               g.setColor(Color.white);

               g.drawString(s,x,20);

               g.setColor(Color.black);

            }

         }

      }

      catch(InterruptedException e) {}

   }

}
